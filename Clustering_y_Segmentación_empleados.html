def asignar_region_base(base):
    if base in ['BOG', 'MDE', 'CLO', 'BGA', 'PEI', 'AXM', 'IBE', 'MZL', 'PSO', 'NVA', 'CUC', 'IPI']:
        return 'Regi√≥n Andina'

        return 'Costa Caribe'
    elif base in ['LET', 'VVC', 'EYP',]:
        return 'Amazon√≠a/Orinoqu√≠a'
    elif base == 'UIB':
        return 'Regi√≥n Pac√≠fica'
    elif base == 'ADZ':
        return 'San Andr√©s y Providencia'
    elif base == 'FLA':
        return 'Regi√≥n Sur'
    else:
        return 'Internacional'
